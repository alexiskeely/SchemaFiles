   a sh:NodeShape ;
   sh:target [
           a sh:SPARQLTarget ;
           sh:prefixes rdfs: ;
           sh:select """
           SELECT ?this
           WHERE {
                   ?this rdfs:label ?name
                   FILTER regex(?name, "/near/", "i")
           }
           """ ;
   ] ;
   a vra:Place ;
